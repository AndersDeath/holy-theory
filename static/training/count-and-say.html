function countAndSay(n: number): string {
    if (n === 1) {
        return "1";
    }

    const prevResult = countAndSay(n - 1);
    let result = "";
    let count = 1;

    for (let i = 0; i < prevResult.length; i++) {
        if (prevResult[i] === prevResult[i + 1]) {
            count++;
        } else {
            result += count + prevResult[i];
            count = 1;
        }
    }

    return result;
};
