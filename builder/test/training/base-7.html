Base 7

function convertToBase7(num: number): string {
  if (num === 0) {
        return "0";
    }

    let isNegative = num < 0;
    num = Math.abs(num);
    let base7 = "";

    while (num > 0) {
        base7 = (num % 7).toString() + base7;
        num = Math.floor(num / 7);
    }

    if (isNegative) {
        base7 = "-" + base7;
    }

    return base7;
};
