function containsNearbyDuplicate(nums: number[], k: number): boolean {
    const seen: any = new Map();
    for (let i = 0; i < nums.length; ++i) {
      if (i - seen.get(nums[i]) <= k) {
        return true;
      }
        seen.set(nums[i], i);
    }
    return false;
};
