Multiple calls:

Single call: add(1); // 1

Mixed thing: let addTwo = add(2); addTwo; // 2 addTwo + 5; // 7 addTwo(3); // 5 addTwo(3)(5); // 10

function add(x: number): any {
    let sum = x;
    function f(y: number) {
      sum += y;
      return f;
    }
    f.toString = function() {
      return sum;
    };
    return f;
  }
